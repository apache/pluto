66      // creates a locale object from a string representation
67      private Locale createLocale(String locale)
68      {
69          // parse locale String
70          StringTokenizer tokenizer = new StringTokenizer(locale,"_");
71          String[] localeDef = new String[5]; // just in case we have more than one variant
72          for (int i = 0; i < localeDef.length; i++)
73          {
74              if (tokenizer.hasMoreTokens())
75              {
76                  localeDef[i] = tokenizer.nextToken();
77              }
78              else
79              {
80                  localeDef[i] = "";
81              }
82          }
83  
84          return new java.util.Locale(localeDef[0], localeDef[1], localeDef[2]+localeDef[3]+localeDef[4]);
85      }

89      public Language get(Locale locale)
90      {
91          if (resources!=null && resourceBundleInitialized==false)
92          {
93              initResourceBundle();
94              this.resourceBundleInitialized = true;
95          }
96  
97          if (!locales.contains(locale))
98          {
99              locale = matchLocale(locale);
100         }
101 
102         Iterator iterator = this.iterator();
103         while (iterator.hasNext())
104         {
105             Language language = (Language)iterator.next();
106             if (language.getLocale().equals(locale) || size()==1)
107             {
108                 return language;
109             }
110         }
111 
112         return null;
113     }

120     public Locale getDefaultLocale()
121     {
122         Locale defLoc = null;
123 
124         if (locales != null && locales.size() > 0)
125         {
126             defLoc = (Locale)locales.firstElement();
127 
128             if (defLoc == null)
129             {
130                 defLoc = new Locale("en","");
131                 locales.add(defLoc);
132             }
133         }
134         else
135         {
136             defLoc = new Locale("en","");
137             locales.add(defLoc);
138         }
139 
140         return defLoc;
141     }

173     private void initInlinedInfos() throws Exception
174     {   
175         // if resource-bundle is given
176         // must be initialized later when classloader is known by initResourceBundle()            
177 
178         if (locales.isEmpty())
179         {
180             getDefaultLocale(); //the defualt gets automaticaly added to the locals
181         }
182         if (castorKeywords == null)
183         {
184             castorKeywords="";
185         }
186         if (shortTitle == null)
187         {
188             shortTitle="";
189         }
190         if (title == null)
191         {
192             title="";
193         }
194         boolean added = add(createLanguage(getDefaultLocale(), null));
195     }

197     // create and add all resource bundle information as Language objects to this set
198     private void initResourceBundle()
199     {
200         Iterator iter = locales.iterator();
201         while (iter.hasNext())
202         {
203             Locale locale = (Locale)iter.next();
204             ResourceBundle bundle = null;
205             bundle = loadResourceBundle(locale);
206             if (bundle != null)
207             {
208                 Language language = createLanguage(locale, bundle);
209                 remove(language);
210                 add(language);
211             }
212         }
213     }

215     // try to match the given locale to a supported locale
216     private Locale matchLocale(Locale locale)
217     {
218 
219         String variant = locale.getVariant();
220         if (variant != null && variant.length() > 0)
221         {
222             locale = new Locale(locale.getLanguage(), locale.getCountry());                                
223         }
224 
225         if (! locales.contains(locale))
226         {
227             String country = locale.getCountry();
228             if (country != null && country.length() > 0)
229             {
230                 locale = new Locale(locale.getLanguage(), "");
231             }
232         }
233 
234         if (! locales.contains(locale))
235         {
236             locale = getDefaultLocale();
237         }
238 
239         return locale;
240     }

267     // loads resource bundle files from WEB-INF/classes directory
268     protected ResourceBundle loadResourceBundle(Locale locale)
269     {
270         ResourceBundle resourceBundle = null;
271         try
272         {
273             if (classLoader != null)
274             {
275                 resourceBundle=ResourceBundle.getBundle(resources, locale, classLoader);
276             }
277             else
278             {
279                 resourceBundle=ResourceBundle.getBundle(resources, locale, Thread.currentThread().getContextClassLoader());
280             }
281         }
282         catch (MissingResourceException x)
283         {
284             return null;
285         }
286         return resourceBundle;
287     }
