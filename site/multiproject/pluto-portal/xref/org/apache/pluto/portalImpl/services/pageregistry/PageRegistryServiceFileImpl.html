27  import javax.servlet.ServletConfig;

40  /***
41   * A simple XML Castor file based implementation of the <code>PageRegistryService</config>
42   * <p>This store persit the PageRegistry informations</p>
43   *
44   */
45  
46  public class PageRegistryServiceFileImpl extends PageRegistryService
47  {
48  
49  
50      // default configuration values
51      public final static String CONFIG_FILE              = "WEB-INF/data/pageregistry.xml";
52      public final static String DEFAULT_MAPPING          = "WEB-INF/data/xml/pageregistrymapping.xml";
53      // configuration keys
54      private final static String CONFIG_MAPPING          = "mapping.configfile";
55  
56      // Castor mapping file
57      private Mapping mapping = null;
58      // Servlet Context
59      private ServletContext servletContext = null;
60      // Root element
61      private PortalImpl registry = null;
62      private RootFragment root = null;
63      private Logger log = null;
64  
65      private HashMap fragments = new HashMap();
66  
67      public void init (ServletConfig config, Properties properties) throws Exception
68      {        
69          servletContext = config.getServletContext();
70          log = Log.getService().getLogger(getClass());
71  
72          String _mapping = properties.getString(CONFIG_MAPPING, DEFAULT_MAPPING);
73          File f = new File(_mapping);
74          if (!f.isAbsolute())
75              _mapping = servletContext.getRealPath(_mapping);
76  
77          this.mapping = new Mapping();
78          try
79          {
80              this.mapping.loadMapping(_mapping);
81          }
82          catch (Exception e)
83          {
84              log.error("Failed to load mapping file "+_mapping,e);
85              throw e;
86          }
87  
88          load();
89  
90          if (log.isDebugEnabled())
91          {
92              log.debug("Dumping complete object model description as it is read from the xml file...");
93              log.debug(registry.toString());
94          }        
95      }
96  
97      public void postInit(ServletConfig config) throws Exception {
98          
99          root = registry.build(config);
100              
101         if (log.isDebugEnabled())
102         {
103             log.debug("Dumping complete navigation tree created of the object model...");
104             log.debug(root.getNavigation().toString());
105         }
106     }
107 
108     public RootFragment getRootFragment()
109     {
110         return root;
111     }
112 
113     public Fragment getFragment(String id) {
114         return (Fragment)fragments.get(id);
115     }
116 
117     public void addFragment(Fragment fragment) throws Exception {
118         String id = fragment.getId();
119         if(! fragments.containsKey(id)) {
120             
121             fragments.put(id, fragment);
122 
123         } else {
124 
125             String msg = "Fragment with this name "+id+" already exists in the pageregistry.xml.";
126             log.error(msg);
127             throw new Exception(msg);
128         }
129 
130     }
131 
132     private void store() throws Exception
133     {
134         String filename = CONFIG_FILE;
135         
136         File f = new File(filename);
137         if (!f.isAbsolute())
138            filename = servletContext.getRealPath(filename);
139         
140         FileWriter writer = new FileWriter(filename);
141         
142         Marshaller marshaller = new Marshaller(writer);
143         
144         marshaller.setMapping(this.mapping);
145         
146         marshaller.marshal(registry);
147     }
148 
149     private void load() throws Exception
150     {
151         String filename = CONFIG_FILE;
152         
153         File f = new File(filename);
154         if (!f.isAbsolute())
155            filename = servletContext.getRealPath(filename);
156 
157         Unmarshaller unmarshaller = new Unmarshaller(this.mapping);
158 
159         registry = (PortalImpl)unmarshaller.unmarshal(new FileReader(filename));
160     }
