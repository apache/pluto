58      public void init(String uniqueContainerName,
59                       ServletConfig servletConfig,
60                       PortletContainerEnvironment environment,
61                       Properties properties)
62      throws PortletContainerException
63      {
64          this.servletConfig = servletConfig;
65          this.uniqueContainerName = uniqueContainerName;
66          PortletContainerServices.createReference(uniqueContainerName, environment);
67          this.properties = properties;
68          initialized = true;
69  
70          // Initialize the Logger that we will use
71          // from here forward for this Container:
72          log = ((LogService)environment
73                  .getContainerService(LogService.class))
74                  .getLogger(getClass());
75      }

83      public void renderPortlet(PortletWindow portletWindow,
84                                HttpServletRequest servletRequest,
85                                HttpServletResponse servletResponse )
86      throws PortletException, IOException, PortletContainerException
87      {
88          PortletContainerServices.prepare(uniqueContainerName);
89          PortletInvoker invoker = null;
90          
91          if (log.isDebugEnabled())
92          {
93              log.debug("PortletContainerImpl.portletService(" + portletWindow.getId() + ") called.");
94          }
95  
96          try
97          {
98              RenderRequest renderRequest = PortletObjectAccess.getRenderRequest(portletWindow, 
99                                                                                 servletRequest, 
100                                                                                servletResponse);
101 
102             RenderResponse renderResponse = PortletObjectAccess.getRenderResponse(portletWindow, 
103                                                                                   servletRequest, 
104                                                                                   servletResponse);
105 
106             invoker = PortletInvokerAccess.getPortletInvoker(portletWindow.getPortletEntity().getPortletDefinition());
107             invoker.render(renderRequest, renderResponse);
108         }
109         finally
110         {
111             PortletInvokerAccess.releasePortletInvoker(invoker);            
112             PortletContainerServices.release();
113         }

118     public void processPortletAction(PortletWindow portletWindow,
119                                      HttpServletRequest servletRequest,
120                                      HttpServletResponse servletResponse)
121     throws PortletException, IOException, PortletContainerException
122     {
123         PortletContainerServices.prepare(uniqueContainerName);
124         PortletInvoker invoker = null;
125 
126         if (log.isDebugEnabled())
127         {
128             log.debug("PortletContainerImpl.performPortletAction(" + portletWindow.getId() + ") called.");
129         }
130 
131         String location = null;
132 
133         InternalActionResponse _actionResponse = null;
134         ActionRequest actionRequest = null;
135 
136         try
137         {
138             /*ActionRequest*/ 
139             actionRequest = 
140             PortletObjectAccess.getActionRequest(portletWindow, 
141                                                  servletRequest, 
142                                                  servletResponse);
143 
144             ActionResponse actionResponse = PortletObjectAccess.getActionResponse(portletWindow, 
145                                                                                   servletRequest, 
146                                                                                   servletResponse);
147             invoker = PortletInvokerAccess.getPortletInvoker(portletWindow.getPortletEntity().getPortletDefinition());
148 
149             _actionResponse = (InternalActionResponse)actionResponse;
150 
151             // call action() at the portlet
152             invoker.action(actionRequest, actionResponse);
153 
154             location = _actionResponse.getRedirectLocation();
155         }
156         catch (PortletException e)
157         {
158             throw e;
159         }
160         catch (IOException e)
161         {
162             throw e;
163         }
164         catch (RuntimeException e)
165         {
166             throw e;
167         }
168         finally
169         {
170             try
171             {
172 				redirect(location, portletWindow, servletRequest, servletResponse, _actionResponse);
173             }
174             finally
175             {
176                 PortletInvokerAccess.releasePortletInvoker(invoker);                            
177                 PortletContainerServices.release();
178             }
179         }
180     }

220     protected void redirect(String location, 
221                               PortletWindow portletWindow, 
222                               HttpServletRequest servletRequest, 
223                               HttpServletResponse servletResponse, 
224                               InternalActionResponse _actionResponse) 
225     throws IOException {
226 	    if (location == null && _actionResponse != null)
227 	    {
228 	        DynamicInformationProvider provider = InformationProviderAccess.getDynamicProvider(servletRequest);
229 	
230 	        //TODO: don't send changes in case of exception -> PORTLET:SPEC:17
231 	
232 	        // get the changings of this portlet entity that might be set during action handling
233 	        // change portlet mode
234 	        if (_actionResponse.getChangedPortletMode() != null)
235 	        {
236 	            InformationProviderAccess.getDynamicProvider(servletRequest)
237 	                .getPortletActionProvider(portletWindow)
238 	                .changePortletMode(_actionResponse.getChangedPortletMode());
239 	        }
240 	        // change window state
241 	        if (_actionResponse.getChangedWindowState() != null)
242 	        {
243 	            InformationProviderAccess.getDynamicProvider(servletRequest)
244 	                .getPortletActionProvider(portletWindow)
245 	                .changePortletWindowState(_actionResponse.getChangedWindowState());
246 	        }
247 	        // get render parameters
248 	        Map renderParameter = _actionResponse.getRenderParameters();
249 	
250 	        PortletURLProvider redirectURL = provider.getPortletURLProvider(portletWindow);
251 	
252 	        if (provider.getPortletMode(portletWindow)!=null)
253 	        {
254 	            redirectURL.setPortletMode(provider.getPortletMode(portletWindow));
255 	        }
256 	        if (provider.getWindowState(portletWindow)!=null)
257 	        {
258 	            redirectURL.setWindowState(provider.getWindowState(portletWindow));
259 	        }
260 	        //            if (secure)
261 	        //            {
262 	        //                redirectURL.setSecure(); // TBD
263 	        //            }
264 	        redirectURL.clearParameters();
265 	        redirectURL.setParameters(renderParameter);
266 	
267 	        location = servletResponse.encodeRedirectURL(redirectURL.toString());
268 	    }
269 	
270 	    javax.servlet.http.HttpServletResponse redirectResponse = servletResponse;
271 	    while (redirectResponse instanceof javax.servlet.http.HttpServletResponseWrapper)
272 	    {
273 	        redirectResponse = (javax.servlet.http.HttpServletResponse)
274 	                           ((javax.servlet.http.HttpServletResponseWrapper)redirectResponse).getResponse();
275 	    }
276 	    redirectResponse.sendRedirect(location);
277 	
278 	    }
