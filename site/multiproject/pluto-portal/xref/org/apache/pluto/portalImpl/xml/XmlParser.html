39      public static org.w3c.dom.Document parsePortletXml(InputStream portletXml) throws IOException, SAXException
40      {
41          
42          DocumentBuilderFactory documentBuilderFactory = DocumentBuilderFactory.newInstance();
43  
44          documentBuilderFactory.setAttribute("http://xml.org/sax/features/validation", Boolean.TRUE);
45          documentBuilderFactory.setAttribute("http://apache.org/xml/features/validation/schema", Boolean.TRUE);
46  
47          documentBuilderFactory.setAttribute("http://xml.org/sax/features/namespaces", Boolean.TRUE);
48          documentBuilderFactory.setAttribute("http://apache.org/xml/features/dom/include-ignorable-whitespace", Boolean.FALSE);
49  
50  
51          try{
52  
53              // throws ParserConfigurationException if documentBuilder cannot be created
54              DocumentBuilder documentBuilder = documentBuilderFactory.newDocumentBuilder(); 
55              documentBuilder.setErrorHandler(new ErrorHandler());
56              documentBuilder.setEntityResolver(new EntityResolver(Constants.RES_PORTLET_DTDS, Constants.RES_PORTLET_DTD_NAMES)); 
57  
58              Document returnDoc = documentBuilder.parse( portletXml );
59              returnDoc.normalize();
60  
61              return returnDoc;
62  
63          } catch (ParserConfigurationException e)
64          {
65              throw new SAXException("Failed creating DocumentBuilder",e);
66          }
67      }

69      public static org.w3c.dom.Document parseWebXml(InputStream webXml) throws IOException, SAXException
70      {
71          DOMParser domParser = new DOMParser();
72  
73          domParser.setErrorHandler(new ErrorHandler());
74          domParser.setEntityResolver(new EntityResolver(Constants.RES_WEB_PUBLIC_ID,
75                                                         Constants.RES_WEB_DTD,
76                                                         Constants.RES_WEB_DTD_NAME));
77              
78          domParser.setFeature("http://xml.org/sax/features/validation", true);
79          domParser.setFeature("http://apache.org/xml/features/dom/include-ignorable-whitespace", false);
80  
81          InputSource source = new InputSource( webXml );
82          
83          domParser.parse( source );
84          
85          return domParser.getDocument();
86      }

106     public static class EntityResolver implements org.xml.sax.EntityResolver
107     {
108 
109         
110         public String publicDTD         = null;
111         public String[] resourceDTDs    = new String[1];
112         public String[] resourceDTDNames= new String[1];
113 
114         public EntityResolver(String publicDTD,
115                               String resourceDTD,
116                               String resourceDTDName)
117         {
118             this.publicDTD       = publicDTD;
119             this.resourceDTDs[0]     = resourceDTD;
120             this.resourceDTDNames[0] = resourceDTDName;
121         }
122 
123         public EntityResolver(String[] resourceDTDs,
124                               String[] resourceDTDNames)
125         {
126             this.resourceDTDs     = resourceDTDs;
127             this.resourceDTDNames = resourceDTDNames;
128         }
129 
130         public InputSource resolveEntity(String publicId, String systemId) throws SAXException 
131         {
132 
133             for (int i=0; i<resourceDTDNames.length; i++)
134             {
135                 if ((publicId != null && publicId.equals(publicDTD))
136                     || (systemId != null && systemId.endsWith(resourceDTDNames[i])))
137                 {
138                     java.io.InputStream is = getClass().getResourceAsStream(resourceDTDs[i]);
139 
140                     if (is != null)
141                         return new InputSource(is);
142                     throw new SAXException("XML configuration DTD not found: "+resourceDTDs[i]);
143                 }
144             }
145             
146             // no other external entity permitted
147             throw new SAXException("External entites are not permitted in XML configuration files");
148             
149         }
150 
151     }
