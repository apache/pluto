87      {
88          return (service.isErrorEnabled (aComponent));

99      }

101     public static void info (String aComponent, String aMessage)
102     {

104     }
105 
106     public static void warn (String aComponent, String aMessage)
107     {
108         service.warn (aComponent, aMessage);
109     }
110 
111     public static void error (String aComponent, String aMessage)
112     {
113         service.error (aComponent, aMessage, null);
114     }
115 
116     public static void error (String aComponent, String aMessage, Throwable aThrowable)
117     {
118         service.error (aComponent, aMessage, aThrowable);
119     }
120 
121     public static void error (String aComponent, Throwable aThrowable)
122     {
123         service.error (aComponent, aThrowable);
124     }

126     private final static LogService service =
