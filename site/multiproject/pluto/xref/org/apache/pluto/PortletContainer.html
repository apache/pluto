16  package org.apache.pluto;
17  
18  import java.util.Properties;
19  import java.io.IOException;
20  
21  import javax.portlet.PortletException;
22  import javax.servlet.ServletConfig;
23  import javax.servlet.http.HttpServletRequest;
24  import javax.servlet.http.HttpServletResponse;
25  
26  import org.apache.pluto.om.window.PortletWindow;
27  import org.apache.pluto.services.PortletContainerEnvironment;
28  
29  /***
30   * The <CODE>PortletContainer</CODE> interface is the entrance point of the portlet
31   * container. This singleton is normally called by the aggregation in a specific
32   * order, meaning that each method of this singleton has to be called in a
33   * defined way.<P>
34   * The base functionality of the portlet container can be enhanced or even
35   * modified by PortletContainerServices.
36   * <P>
37   * The methods of this class have to be called in the following order:
38   * <UL>
39   * <LI>only once</LI>
40   *   <UL>
41   *   <LI>init</LI>
42   *   <LI>destroy</LI>
43   *   </UL>
44   * <LI>for each request</LI>
45   *   <UL>
46   *   <LI>renderPortlet (for each portlet)</LI>
47   *   </UL>
48   * <LI>optional for each request</LI>
49   *   <UL>
50   *   <LI>processPortletAction (for portlet receiving the action request)</LI>
51   *   </UL>
52   * </UL>
53   * 
54   * @version $Id: PortletContainer.html,v 1.1 2004/03/04 13:43:20 cziegeler Exp $
55   */
56  public interface PortletContainer
57  {
58  
59      /***
60       * Initializes the portlet container.
61       * 
62       * @param uniqueContainerName container name that must be unqiue across all containers defined within this JVM. This name must be identical across JVMs.
63       * @param servletConfig the servlet configuration
64       * @param environment the portlet container environment including all services
65       * @param properties the portlet container specific properties may vary from container to container
66       * @exception PortletContainerException
67       *                if an error occurs while initializing the container
68       */
69      public void init(String uniqueContainerName,
70                       ServletConfig servletConfig,
71                       PortletContainerEnvironment environment,
72                       Properties properties) throws PortletContainerException;
73  
74      /***
75       * Shuts down the portlet container.
76       * After calling this method it is no longer valid to call any method on the portlet container.
77       * 
78       * @exception PortletContainerException
79       *                if an error occurs while shutting down the container
80       */
81      public void shutdown() throws PortletContainerException;
82  
83      /***
84       * Calls the render method of the given portlet window.
85       * 
86       * @param portletWindow
87       *                 the portlet Window
88       * @param request  the servlet request
89       * @param response the servlet response
90       * @exception PortletException
91       *                   if one portlet has trouble fulfilling the request
92       * @exception IOException
93       *                   if the streaming causes an I/O problem
94       * @exception PortletContainerException
95       *                   if the portlet container implementation has trouble fulfilling the request
96       */
97      public void renderPortlet(PortletWindow portletWindow,
98                                HttpServletRequest request,
99                                HttpServletResponse response)
100     throws PortletException, IOException, PortletContainerException;
101 
102 
103     /***
104      * Indicates that a portlet action occured in the current request and
105      * calls the processAction method of this portlet.
106      * 
107      * @param portletWindow
108      *                 the portlet Window
109      * @param request  the servlet request
110      * @param response the servlet response
111      * @exception PortletException
112      *                   if one portlet has trouble fulfilling the request
113      * @exception PortletContainerException
114      *                   if the portlet container implementation has trouble fulfilling the request
115      */
116     public void processPortletAction(PortletWindow portletWindow,
117                                      HttpServletRequest request,
118                                      HttpServletResponse response)
119     throws PortletException, IOException, PortletContainerException;
120 
121     /***
122      * Indicates that the portlet must be initialized
123      * 
124      * @param portletWindow
125      *                 the portlet Window
126      * @param request  the servlet request
127      * @param response the servlet response
128      * @exception PortletException
129      *                   if one portlet has trouble fulfilling the request
130      * @exception PortletContainerException
131      *                   if the portlet container implementation has trouble fulfilling the request
132      */
133     public void portletLoad(PortletWindow portletWindow,
134                             HttpServletRequest servletRequest,
135                             HttpServletResponse servletResponse )
136     throws PortletException, PortletContainerException;
137 
138     /***
139      * Returns whether the container is already initialized or not.
140      * 
141      * @return <code>true</code> if the container is initialized
142      */
143     public boolean isInitialized();
144 }
