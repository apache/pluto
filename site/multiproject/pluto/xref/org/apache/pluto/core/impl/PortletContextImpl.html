47      public javax.portlet.PortletRequestDispatcher getRequestDispatcher(String path)
48      {
49  		try {
50  	        javax.servlet.RequestDispatcher rd = servletContext.getRequestDispatcher(path);
51      	    return new PortletRequestDispatcherImpl(rd);
52          } catch (Exception e) {
53      		// need to catch exception because of tomcat 4.x bug
54      		// tomcat throws an exception instead of return null
55      		// if the path was not found
56      		return null;
57      	}
58  
59      }
60  
61      public javax.portlet.PortletRequestDispatcher getNamedDispatcher(String name)
62      {
63         	javax.servlet.RequestDispatcher rd = servletContext.getNamedDispatcher(name);
64         	return rd != null ? new PortletRequestDispatcherImpl(rd)
65             	              : null;
66      }

98      public java.net.URL getResource(String path) throws java.net.MalformedURLException
99      {
100         if (path == null || !path.startsWith("/"))
101         {
102             throw new MalformedURLException("path must start with a '/'");
103         }
104         return servletContext.getResource(path);
105     }
