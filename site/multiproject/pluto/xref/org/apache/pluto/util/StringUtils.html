30      static java.util.BitSet dontNeedEncoding;
31      static final int caseDiff = ('a' - 'A');
32  
33      /* The list of characters that are not encoded have been determined by
34         referencing O'Reilly's "HTML: The Definitive Guide" (page 164). */
35  
36      static {
37          dontNeedEncoding = new java.util.BitSet(256);
38          int i;
39          for (i = 'a'; i <= 'z'; i++)
40          {
41              dontNeedEncoding.set(i);
42          }
43          for (i = 'A'; i <= 'Z'; i++)
44          {
45              dontNeedEncoding.set(i);
46          }
47          for (i = '0'; i <= '9'; i++)
48          {
49              dontNeedEncoding.set(i);
50          }
51          dontNeedEncoding.set('-');
52          dontNeedEncoding.set('_');
53          dontNeedEncoding.set('.');
54          dontNeedEncoding.set('*');
55      }

121     /***
122      * Returns a combination of two paths, inserting slashes as appropriate.
123      *
124      * @param   aRoot
125      *          a root path
126      * @param   aPath
127      *          a path
128      *
129      * @return   the path

131 
132     public static String pathOf (String aRoot, String aPath)
133     {
134         if (aPath == null)
135         {
136             throw new IllegalArgumentException ("StringUtils: Argument \"aPath\" cannot be null.");
137         }
138 
139         String result = null;
140 
141         if (aPath.startsWith ("/")  ||
142             aPath.startsWith ("//") ||
143             (aPath.length () >= 2 && aPath.charAt (1) == ':'))
144         {
145             result = aPath;
146         }
147         else
148         {
149             if (aRoot == null)
150             {
151                 throw new IllegalArgumentException ("StringUtils: Argument \"aRoot\" cannot be null.");
152             }
153 
154             StringBuffer temp = new StringBuffer (aRoot);
155 
156             if (! aRoot.endsWith ("/") &&
157                 ! aRoot.endsWith ("//"))
158             {
159                 temp.append ('/');
160             }
161 
162             temp.append (aPath);
163 
164             result = temp.toString ();
165         }
166 
167         return result.toString();
168     }

170     /***
171      * Returns a <CODE>Boolean</CODE> object that corresponds the given value.
172      * A value of <CODE>true</CODE> or <CODE>yes</CODE> corresponds to
173      * <CODE>Boolean.TRUE</CODE> and a value of <CODE>false</CODE> or
174      * <CODE>no</CODE> corresponds to <CODE>Boolean.FALSE</CODE>.
175      * The comparions is case-insensitive, but for performance reasons,
176      * lower-case values of <CODE>true</CODE> and <CODE>false</CODE>
177      * should be used.
178      *
179      * @param   aValue
180      *          to value to convert
181      *
182      * @return   the boolean value
183      */
184     public static Boolean booleanOf (String aValue)
185     {
186         Boolean result = null;
187 
188         if (aValue != null)
189         {
190             if (aValue == "true" ||
191                 aValue == "yes" ||
192                 aValue.equalsIgnoreCase ("true") ||
193                 aValue.equalsIgnoreCase ("yes"))
194             {
195                 result = Boolean.TRUE;
196             }
197             else if (aValue == "false" ||
198                      aValue == "no" ||
199                      aValue.equalsIgnoreCase ("false") ||
200                      aValue.equalsIgnoreCase ("no"))
201             {
202                 result = Boolean.FALSE;
203             }
204         }
205 
206         return(result);
207     }

209     /***
210      * Replace all occurrences of a pattern within a string by a replacement
211      *
212      * @param source  The string that should be searched
213      * @param pattern  The pattern that should be replaced
214      * @param replace  The replacement that should be inserted instead of the pattern
215      *
216      * @return  The updated source string
217      */
218     public static String replace(String source, String pattern, String replace)
219     {
220         if (source == null || source.length() == 0 ||
221             pattern == null || pattern.length() == 0)
222         {
223             return source;
224         }
225 
226         int k = source.indexOf(pattern);
227 
228         if (k == -1)
229         {
230             return source;
231         }
232 
233         StringBuffer out = new StringBuffer();
234         int i=0, l=pattern.length();
235 
236         while (k != -1)
237         {
238             out.append(source.substring(i,k));
239 
240             if (replace != null) {
241                 out.append(replace);
242             }
243 
244             i = k+l;
245             k = source.indexOf(pattern, i);
246         }
247         out.append(source.substring(i));
248         return out.toString();
249     }

262     public static String encode(String s)
263     {
264         int maxBytesPerChar = 10;
265         StringBuffer out = new StringBuffer(s.length());
266         java.io.ByteArrayOutputStream buf = new java.io.ByteArrayOutputStream(maxBytesPerChar);
267         java.io.OutputStreamWriter writer = new java.io.OutputStreamWriter(buf);
268 
269         for (int i = 0; i < s.length(); i++)
270         {
271             int c = (int)s.charAt(i);
272             if (dontNeedEncoding.get(c))
273             {
274                 out.append((char)c);
275             }
276             else
277             {
278                 // convert to external encoding before hex conversion
279                 try
280                 {
281                     writer.write(c);
282                     writer.flush();
283                 }
284                 catch (java.io.IOException e)
285                 {
286                     buf.reset();
287                     continue;
288                 }
289                 byte[] ba = buf.toByteArray();
290                 for (int j = 0; j < ba.length; j++)
291                 {
292                     out.append('x');
293                     char ch = Character.forDigit((ba[j] >> 4) & 0xF, 16);
294                     // converting to use uppercase letter as part of
295                     // the hex value if ch is a letter.
296                     if (Character.isLetter(ch))
297                     {
298                         ch -= caseDiff;
299                     }
300                     out.append(ch);
301                     ch = Character.forDigit(ba[j] & 0xF, 16);
302                     if (Character.isLetter(ch))
303                     {
304                         ch -= caseDiff;
305                     }
306                     out.append(ch);
307                 }
308                 buf.reset();
309             }
310         }
311 
312         return out.toString();
313     }

315     public static String decode(String s)
316     {
317         StringBuffer sb = new StringBuffer();
318         for (int i=0; i<s.length(); i++)
319         {
320             char c = s.charAt(i);
321             switch (c)
322             {
323                 case '%':
324                     if (((s.charAt(i+1)>='0') && (s.charAt(i+1)<='9')) &&
325                         ((s.charAt(i+2)>='0') && (s.charAt(i+2)<='9')))
326                     {
327                         try
328                         {
329                             sb.append((char)Integer.parseInt(s.substring(i+1, i+3), 16));
330                         }
331                         catch (java.lang.NumberFormatException e)
332                         {
333                             throw new java.lang.IllegalArgumentException();
334                         }
335                         i += 2;
336                         break;
337                     }
338                 default:
339                     sb.append(c);
340                     break;
341             }
342         }
343         // Undo conversion to external encoding
344         String result = sb.toString();
345         try
346         {
347             byte[] inputBytes = result.getBytes("8859_1");
348             result = new String(inputBytes);
349         }
350         catch (java.io.UnsupportedEncodingException e)
351         {
352             // The system should always have 8859_1
353         }
354         return result;
355     }
