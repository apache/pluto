73      public Map getParameterMap() 
74      {
75          //get control params
76          Map portletParameters = new HashMap();
77               
78          Iterator iterator = control.getRenderParamNames(portletWindow);
79          while (iterator.hasNext())
80          {
81              String name = (String)iterator.next();
82  
83              String[] values = control.getRenderParamValues(portletWindow, name);
84  
85              portletParameters.put(name, values );
86  
87          }
88  
89          //get request params      
90          String pid = control.getPIDValue();
91          String wid = portletWindow.getId().toString();
92          if (pid.equals(wid)) {
93              for (Enumeration parameters = super.getParameterNames(); parameters.hasMoreElements();) {
94                  String   paramName   = (String)parameters.nextElement();
95                  String[] paramValues = (String[])super.getParameterValues(paramName);
96                  String[] values      = (String[])portletParameters.get(paramName);
97      
98                  if (values != null) {
99                      String[] temp = new String[paramValues.length + values.length];
100                     System.arraycopy(paramValues, 0, temp, 0, paramValues.length);
101                     System.arraycopy(values, 0, temp, paramValues.length, values.length);
102                     paramValues = temp;
103                 }
104                 portletParameters.put(paramName, paramValues);
105             }
106         }
107 
108         return Collections.unmodifiableMap(portletParameters);
109     }
110     
111     public Enumeration getParameterNames() 
112     {
113         return Collections.enumeration(this.getParameterMap().keySet());
114     }
