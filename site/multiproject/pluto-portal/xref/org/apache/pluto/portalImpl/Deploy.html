74      public static void deployArchive(String webAppsDir, String warFile)
75          throws IOException {
76          String warFileName = warFile;
77          if (warFileName.indexOf("/") != -1)
78              warFileName =
79                  warFileName.substring(warFileName.lastIndexOf("/") + 1);
80          if (warFileName.indexOf(dirDelim) != -1)
81              warFileName =
82                  warFileName.substring(warFileName.lastIndexOf(dirDelim) + 1);
83          if (warFileName.endsWith(".war"))
84              warFileName =
85                  warFileName.substring(0, warFileName.lastIndexOf("."));
86  
87          System.out.println("deploying '" + warFileName + "' ...");
88  
89          String destination = webAppsDir + warFileName;
90  
91          JarFile jarFile = new JarFile(warFile);
92          Enumeration files = jarFile.entries();
93          while (files.hasMoreElements()) {
94              JarEntry entry = (JarEntry) files.nextElement();
95  
96              File file = new File(destination, entry.getName());
97              File dirF = new File(file.getParent());
98              dirF.mkdirs();
99              if (entry.isDirectory()) {
100                 file.mkdirs();
101             } else {
102                 byte[] buffer = new byte[1024];
103                 int length = 0;
104                 InputStream fis = jarFile.getInputStream(entry);
105                 FileOutputStream fos = new FileOutputStream(file);
106                 while ((length = fis.read(buffer)) >= 0) {
107                     fos.write(buffer, 0, length);
108                 }
109                 fos.close();
110             }
111 
112         }
113 
114         System.out.println("finished!");
115     }

487     static private void addToEntityReg(String[] args) {
488         String pluto = args[1];
489         File portletAppFile = new File(args[2]);
490         String portletAppFileName = portletAppFile.getName();
491         String portletApp =
492             portletAppFileName.substring(0,	portletAppFileName.lastIndexOf(".war"));
493         int o = (args[4].equals("-addToEntityReg") ? 5 : 6);
494         String appId = args[o++];
495         try {
496             String entityMapping = webAppsDir + portalImplWebDir
497   					               + "WEB-INF/data/portletentityregistry.xml";
498             File file = new File(entityMapping);
499             RandomAccessFile ras = new RandomAccessFile(file, "rw");
500             long length = ras.length();
501             byte[] contentByte = new byte[(int) length];
502             ras.read(contentByte);
503             String contentString = new String(contentByte);
504             long pos = contentString.lastIndexOf("</portlet-entity-registry>");
505             ras.seek(pos);
506             ras.writeBytes("    <application id=\"" + appId + "\">\r\n");
507             ras.writeBytes("        <definition-id>" + portletApp + "</definition-id>\r\n");
508 
509             StringTokenizer tokenizer;
510             for (int i = o; i < args.length; ++i) {
511                 tokenizer = new StringTokenizer(args[i], ":");
512                 String portletId = tokenizer.nextToken();
513                 String portletName = tokenizer.nextToken();
514                 ras.writeBytes("        <portlet id=\"" + portletId + "\">\r\n");
515                 ras.writeBytes("            <definition-id>" + portletApp
516                                + "." + portletName + "</definition-id>\r\n");
517                 ras.writeBytes("        </portlet>\r\n");
518             }
519             ras.writeBytes("    </application>\r\n");
520             ras.writeBytes("</portlet-entity-registry>\r\n");
521             ras.close();
522 
523         } catch (Exception e) {
524             e.printStackTrace(System.out);
525         }
526     }
