65  /***
66   * <p>Responsible for rendering a single Portlet.<p>
67   *
68   * <p>Requires two JSP files to exist, PortletFragmentHeader.jsp
69   * and PortletFragmentFooter.jsp. These pages define the header
70   * and footer of the portlet.</p>
71   *
72   * <p>A PortletFragment.PortletInfo object is added to the request
73   * attributes and makes information about the Portlet available to the
74   * header and footer JSP pages.</p>
75   *
76   * @author Stephan Hesmer
77   * @author Nick Lothian
78   *
79   */

92              throws Exception

95          log = Log.getService().getLogger(getClass());
96  		String portletEntityId = getInitParameters().getString("portlet");
97  		PortletEntity portletEntity = PortletEntityRegistry.getPortletEntity(ObjectID.createFromString(portletEntityId));

100 		PortletWindowList windowList = portletEntity.getPortletWindowList();

104 	public void service(HttpServletRequest request, HttpServletResponse response)
105             throws ServletException, IOException
106     {
107 		HttpServletRequest wrappedRequest = ServletObjectAccess.getServletRequest(request, portletWindow);
108 		PrintWriter responseWriter = response.getWriter();
109 		StringWriter storedWriter = new StringWriter();
110 
111 		// load the Portlet
112 		// If there is an error loading, then we will save the error message and attempt
113 		// to render it inside the Portlet, so the Portal has a chance of still looking
114 		// okay
115 		String errorMsg = null;
116 		try {
117 			PortletContainerFactory.getPortletContainer().portletLoad(portletWindow, wrappedRequest, response);
118 		} catch (PortletContainerException e) {
119 			log.error("Error in Portlet", e);
120 			errorMsg = getErrorMsg();
121 		} catch (Throwable t) {
122 			// If we catch any throwable, we want to try to continue
123 			// so that the rest of the portal renders correctly
124 			log.error("Error in Portlet", t);
125 			if (t instanceof VirtualMachineError) {
126 				// if the Throwable is a VirtualMachineError then
127 				// it is very unlikely (!) that the portal is going
128 				// to render correctly.
129 				throw (Error) t;
130 			} else {
131 				errorMsg = getErrorMsg();
132 			}
133 
134 		}
135 
136 		PortalEnvironment env = (PortalEnvironment) request.getAttribute(PortalEnvironment.REQUEST_PORTALENV);
137 		PortalURL thisURL = env.getRequestedPortalURL();
138 
139 		PortalControlParameter thisControl = new PortalControlParameter(thisURL);
140 		if (thisControl.isOnePortletWindowMaximized()) {
141 			WindowState currentState = thisControl.getState(portletWindow);
142 			if (!WindowState.MAXIMIZED.equals(currentState)) {
143 				return;
144 			}
145 		}
146 
147 		PortletDefinition portletDefinition = portletWindow.getPortletEntity().getPortletDefinition();
148 		LanguageSet languageSet = portletDefinition.getLanguageSet();
149 		Language lang = languageSet.get(request.getLocale());
150 		String title = lang != null ? lang.getTitle() : "no title available";
151 
152 		// create a PortletInfo object. This is used to communicate with
153 		// the header and footer JSP pages for this portlet
154 		PortletInfo portletInfo = new PortletInfo();
155 
156 		if (!portletWindow.getPortletEntity().getPortletDefinition().getServletDefinition().isUnavailable()) {
157 			PrintWriter writer2 = new PrintWriter(storedWriter);
158 
159 			// create a wrapped response which the Portlet will be rendered to
160 			HttpServletResponse wrappedResponse = ServletObjectAccess.getStoredServletResponse(response, writer2);
161 
162 			try {
163 				// render the Portlet to the wrapped response, to be output later.
164 				PortletContainerFactory.getPortletContainer().renderPortlet(portletWindow, wrappedRequest, wrappedResponse);
165 			} catch (UnavailableException e) {
166 				log.error("Portlet is Unavailable", e);
167 				writer2.println("the portlet is currently unavailable!");
168 
169 				ServletDefinitionCtrl servletDefinitionCtrl = (ServletDefinitionCtrl) ControllerObjectAccess.get(portletWindow.getPortletEntity().getPortletDefinition().getServletDefinition());
170 				if (e.isPermanent()) {
171 					servletDefinitionCtrl.setAvailable(Long.MAX_VALUE);
172 				} else {
173 					int unavailableSeconds = e.getUnavailableSeconds();
174 					if (unavailableSeconds <= 0) {
175 						unavailableSeconds = 60; // arbitrary default
176 					}
177 					servletDefinitionCtrl.setAvailable(System.currentTimeMillis() + unavailableSeconds * 1000);
178 				}
179 			} catch (Exception e) {
180 				log.error("Error in Portlet", e);
181 				writer2.println(getErrorMsg());
182 			}
183 			String dyn_title = ((DynamicTitleServiceImpl) FactoryAccess.getDynamicTitleContainerService()).getDynamicTitle(portletWindow, request);
184 			if (dyn_title != null) {
185 				title = dyn_title;
186 			}
187 
188 		}
189 		// set the title, so the header JSP page can use it when rendering
190 		portletInfo.setTitle(title);
191 
192 		DynamicInformationProvider provider = FactoryAccess.getDynamicProvider(request);
193 		ContentType supported = portletDefinition.getContentTypeSet().get(wrappedRequest.getContentType());
194 		PortalContextProvider portalContextProvider = FactoryAccess.getStaticProvider().getPortalContextProvider();
195 
196 		// get the list of modes this Portlet supports
197 		if (supported != null && portalContextProvider != null) {
198 			// if portlet supports portlet modes
199 			Iterator modes = supported.getPortletModes();
200 			while (modes.hasNext()) {
201 				PortletMode mode = (PortletMode) modes.next();
202 
203 				// check whether portal also supports portlet mode
204 				boolean portalSupport = false;
205 				Iterator portalSupportedModes = portalContextProvider.getSupportedPortletModes().iterator();
206 				while (portalSupportedModes.hasNext()) {
207 					PortletMode portalSupportedMode = (PortletMode) portalSupportedModes.next();
208 					if (mode.equals(portalSupportedMode)) {
209 						portalSupport = true;
210 						break;
211 					}
212 				}
213 
214 				// create links for portlet modes in portlet header
215 				if (portalSupport) {
216 					env = (PortalEnvironment) request.getAttribute(PortalEnvironment.REQUEST_PORTALENV);
217 					PortalURL modeURL = env.getRequestedPortalURL();
218 
219 					PortalControlParameter control = new PortalControlParameter(modeURL);
220 					PortletMode currentMode = control.getMode(portletWindow);
221 					control.setMode(portletWindow, mode);
222 
223 					portletInfo.addPortletMode(mode, modeURL.toString(control, new Boolean(request.isSecure())), mode.equals(currentMode));
224 				}
225 			}
226 
227 			// get the list of window states this Portlet supports
228 			Iterator states = portalContextProvider.getSupportedWindowStates().iterator();
229 			while (states.hasNext()) {
230 				WindowState state = (WindowState) states.next();
231 				env = (PortalEnvironment) request.getAttribute(PortalEnvironment.REQUEST_PORTALENV);
232 				PortalURL stateURL = env.getRequestedPortalURL();
233 				PortalControlParameter control = new PortalControlParameter(stateURL);
234 				WindowState currentState = control.getState(portletWindow);
235 
236 				control.setState(portletWindow, state);
237 				portletInfo.addPortletWindowState(state.toString().substring(0, 3), stateURL.toString(control, new Boolean(request.isSecure())), state.equals(currentState));
238 			}
239 		}
240 		// output the header JSP page
241 		request.setAttribute("portletInfo", portletInfo);
242 		RequestDispatcher rd = getServletConfig().getServletContext().getRequestDispatcher(BASE_ROOT + "PortletFragmentHeader.jsp");
243 		rd.include(request, response);
244 		try {
245 			// output the Portlet
246 			// check if there is an error message
247 			if (errorMsg == null) {
248 				// no error message, so output the Portlet
249 				if (portletWindow.getPortletEntity().getPortletDefinition().getServletDefinition().isUnavailable()) {
250 					// the portlet is unavailable
251 					responseWriter.println("the portlet is currently unavailable!");
252 				} else {
253 					responseWriter.println(storedWriter.toString());
254 				}
255 			} else {
256 				// output the errror message
257 				responseWriter.println(errorMsg);
258 			}
259 		} finally {
260 			// output the footer JSP page
261 			RequestDispatcher rdFooter = getServletConfig().getServletContext().getRequestDispatcher(BASE_ROOT + "PortletFragmentFooter.jsp");
262 			rdFooter.include(request, response);
263 			request.removeAttribute("portletInfo");
264 		}
265 
266 	}
