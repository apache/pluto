70      {
71          // get container service stack out of the thread local
72          Stack currentContainerServiceStack = (Stack)currentContainerService.get();
73          // add current environment to stack stored in thread local

76          // get uniqueContainerName stack out of the thread local
77          Stack currentNameStack = (Stack)currentUniqueContainerName.get();
78          // add current uniqueContainerName to stack stored in thread local
79          currentNameStack.push(uniqueContainerName);
80      }
