119     public PortletSession getPortletSession(boolean create)
120     {
121         // check if the session was invalidated
122         javax.servlet.http.HttpSession httpSession = this._getHttpServletRequest().getSession(false);
123 
124         if ((portletSession != null) && (httpSession == null))
125         {
126         	portletSession = null;
127         }
128         else if (httpSession != null)
129         {
130         	create = true;
131         }
132 
133         if (create && portletSession == null)
134         {
135         	httpSession = this._getHttpServletRequest().getSession(create);
136         	if (httpSession != null)
137         	{
138         		portletSession = PortletObjectAccess.getPortletSession(portletWindow, httpSession);
139         	}
140         }
141 
142         return portletSession;
143     }

145     public String getProperty(String name)
146     {
147         if (name == null)
148         {
149             throw new IllegalArgumentException("Attribute name == null");
150         }
151 
152         // get properties from request header
153         String prop = this._getHttpServletRequest().getHeader(name);
154         if (prop == null)
155         {
156             // get properties from PropertyManager
157             Map map = PropertyManager.getRequestProperties(portletWindow, this._getHttpServletRequest());
158             if (map != null)
159             {
160                 String[] properties = (String[]) map.get(name);
161                 if ((properties != null) && (properties.length > 0))
162                 {
163                     prop = properties[0];
164                 }
165             }
166         }
167 
168         return prop;
169     }

171     public Enumeration getProperties(String name)
172     {
173         if (name == null)
174         {
175             throw new IllegalArgumentException("Property name == null");
176         }
177 
178         Set v = new HashSet();
179 
180         // get properties from request header
181         Enumeration props = this._getHttpServletRequest().getHeaders(name);
182         if (props != null)
183         {
184             while (props.hasMoreElements())
185             {
186                 v.add(props.nextElement());
187             }
188         }
189 
190         // get properties from PropertyManager
191         Map map = PropertyManager.getRequestProperties(portletWindow, this._getHttpServletRequest());
192         if (map != null)
193         {
194             String[] properties = (String[]) map.get(name);
195 
196             if (properties != null)
197             {
198                 // add properties to vector
199                 for (int i=0;i<properties.length;i++)
200                 {
201                     v.add(properties[i]);
202                 }
203             }
204         }
205 
206         return new Enumerator(v.iterator());
207     }

290     public Enumeration getAttributeNames()
291     {
292         Enumeration attributes = this._getHttpServletRequest().getAttributeNames();
293 
294         Vector portletAttributes = new Vector();
295 
296         while (attributes.hasMoreElements())
297         {
298             String attribute = (String)attributes.nextElement();
299 
300             String portletAttribute = NamespaceMapperAccess.getNamespaceMapper().decode(portletWindow.getId(),attribute);
301 
302             if (portletAttribute!=null)
303             { // it is in the portlet's namespace
304                 portletAttributes.add(portletAttribute);
305             }
306         }
307 
308         return portletAttributes.elements();
309     }

362     public void setAttribute(String name, Object o)
363     {
364         if (name == null)
365         {
366             throw new IllegalArgumentException("Attribute name == null");
367         }
368 
369         if ( o == null)
370         {
371             this.removeAttribute(name);
372         }
373         else if (isNameReserved(name))
374         {
375             // Reserved names go directly in the underlying request
376             _getHttpServletRequest().setAttribute(name, o);
377         }
378         else
379         {
380             this._getHttpServletRequest().setAttribute(NamespaceMapperAccess.getNamespaceMapper().encode(portletWindow.getId(),name), o);
381         }
382     }
