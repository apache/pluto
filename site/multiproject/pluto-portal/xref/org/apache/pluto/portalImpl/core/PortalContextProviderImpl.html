103     private Vector getDefaultModes()
104     {
105 
106         Vector m = new Vector();
107 
108         String[] supportedModes = Config.getParameters().getStrings("supported.portletmode");
109 
110         for (int i=0; i<supportedModes.length; i++) {
111             m.add(new PortletMode(supportedModes[i].toString().toLowerCase()));
112         }
113 
114         return m;
115     }
116 
117     private Vector getDefaultStates()
118     {
119         Vector s = new Vector();
120 
121         String[] supportedStates = Config.getParameters().getStrings("supported.windowstate");
122 
123         for (int i=0; i<supportedStates.length; i++) {
124             s.add(new WindowState(supportedStates[i].toString().toLowerCase()));
125         }
126 
127         return s;
128     }
