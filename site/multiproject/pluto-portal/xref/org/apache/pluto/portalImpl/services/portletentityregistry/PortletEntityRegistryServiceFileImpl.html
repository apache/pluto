51  public class PortletEntityRegistryServiceFileImpl extends PortletEntityRegistryService

55      // default configuration values
56      public final static String CONFIG_FILE              = "WEB-INF/data/portletentityregistry.xml";
57      public final static String DEFAULT_MAPPING          = "WEB-INF/data/xml/portletentitymapping.xml";
58      // configuration keys
59      protected final static String CONFIG_MAPPING          = "mapping.configfile";
60  
61      // Castor mapping file
62      protected Mapping mapping = null;
63      // Servlet Context
64      protected ServletContext servletContext = null;
65      private Logger log = null;
66      // Registry
67      protected PortletApplicationEntityListImpl registry = null;
68      
69      // Helper lists and hashtables to access the data as fast as possible
70      protected Map portletEntitiesKeyObjectID = new HashMap();

72      public void init (ServletConfig servletConfig, Properties properties) throws Exception
73      {
74          servletContext = servletConfig.getServletContext();
75          log = Log.getService().getLogger(getClass());
76  
77          String _mapping = properties.getString(CONFIG_MAPPING, DEFAULT_MAPPING);
78          File f = new File(_mapping);
79          if (!f.isAbsolute())
80              _mapping = servletContext.getRealPath(_mapping);
81  
82          this.mapping = new Mapping();
83          try
84          {
85              this.mapping.loadMapping(_mapping);
86          }
87          catch (Exception e)
88          {
89              log.error("Failed to load mapping file "+_mapping,e);
90              throw e;
91          }
92      
93          load();
94      }

101     public PortletEntity getPortletEntity(ObjectID id)

106     public void store() throws IOException
107     {
108         String filename = CONFIG_FILE;
109         
110         File f = new File(filename);
111         if (!f.isAbsolute())
112            filename = servletContext.getRealPath(filename);
113         
114         FileWriter writer = new FileWriter(filename);
115         
116         try {
117             Marshaller marshaller = new Marshaller(writer);
118 
119             marshaller.setMapping(this.mapping);
120 
121             registry.preStore(null);
122 
123             marshaller.marshal(registry);
124 
125             registry.postStore(null);
126         }
127         catch (org.exolab.castor.mapping.MappingException e)
128         {
129             e.printStackTrace(System.err);
130             throw new IOException(e.toString());
131         }
132         catch (org.exolab.castor.xml.ValidationException e)
133         {
134             e.printStackTrace(System.err);
135             throw new IOException(e.toString());
136         }
137         catch (org.exolab.castor.xml.MarshalException e)
138         {
139             e.printStackTrace(System.err);
140             throw new IOException(e.toString());
141         }
142         catch (Exception e)
143         {
144             e.printStackTrace(System.err);
145             throw new IOException(e.toString());
146         }
147     }

149     public void load() throws IOException
150     {
151         _load();
152 
153         if (log.isDebugEnabled())
154         {
155             log.debug("Dumping complete object model description as it is read from the xml file...");
156             log.debug(registry.toString());
157         }
158 
159         fill();
160     }

164     }

166     private void _load() throws IOException
167     {
168         String filename = CONFIG_FILE;
169         
170         File f = new File(filename);
171         if (!f.isAbsolute())
172            filename = servletContext.getRealPath(filename);
173 
174         try {            
175 
176             Unmarshaller unmarshaller = new Unmarshaller(this.mapping);            
177 
178             registry = (PortletApplicationEntityListImpl)unmarshaller.unmarshal(new FileReader(filename));
179 
180             registry.postLoad(null);
181 
182             registry.preBuild(null);
183 
184             registry.postBuild(null);
185         }
186         catch (org.exolab.castor.mapping.MappingException e)
187         {
188             e.printStackTrace(System.err);
189             throw new IOException(e.toString());
190         }
191         catch (org.exolab.castor.xml.ValidationException e)
192         {
193             e.printStackTrace(System.err);
194             throw new IOException(e.toString());
195         }
196         catch (org.exolab.castor.xml.MarshalException e)
197         {
198             e.printStackTrace(System.err);
199             throw new IOException(e.toString());
200         }
201         catch (Exception e)
202         {
203             e.printStackTrace(System.err);
204             throw new IOException(e.toString());
205         }
206 
207     }

209     private void fill()
210     {
211         Iterator iterator = registry.iterator();
212         while (iterator.hasNext())
213         {
214             PortletApplicationEntity appInst = (PortletApplicationEntity)iterator.next();
215 
216             // fill portletEntitiesKeyObjectID
217             Iterator portlets = appInst.getPortletEntityList().iterator();
218             while (portlets.hasNext())
219             {
220                 PortletEntity portletInst = (PortletEntity)portlets.next();                                
221                 portletEntitiesKeyObjectID.put(portletInst.getId().toString(), portletInst);
222 
223             }
224 
225         }
226 
227     }
